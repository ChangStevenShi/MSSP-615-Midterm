label_formatter <- function(x) {
  sapply(x, function(x) {
    if (is.na(x)) {
      return(NA)
    } else if (x >= 1e6) {
      paste0(formatC(x / 1e6, format = "f", digits = 1), "M")
    } else if (x >= 1e3) {
      paste0(formatC(x / 1e3, format = "f", digits = 1), "K")
    } else {
      as.character(x)
    }
  })
}
